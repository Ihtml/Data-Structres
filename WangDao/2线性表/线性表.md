## 静态表的创建

```c++
// 静态分配
#define MaxSize 50
typedef struct 
{
    int data[MaxSize];
    int length;
}SqList;

// 动态分配
#define InitSize 100
typedef struct 
{
    int *data;
    int MaxSize, length;
}SqList;

// C语言初始分配
L.data = (int*)malloc(sizeof(int)*InitSize);
// C++初始动态分配
L.data = new int[InitSize];
```

## 顺序表的基本操作
### 插入操作
```c++
bool ListInsert(SqList &L, int i, int e){
    if(i<1 || i>L.length+1 || L.length>=MaxSize)
        return false;
    for(int j=L.length; j>=i; j--)
        L.data[i-1] = e;
    L.length++;
    return true;
}
```

### 删除操作
```c++
bool ListDelete(SqList &L, int i, int &e){
    if(i<1 || i>L.length)
        return false;
    e=L.data[i-1];
    for(int j=i; j<L.length; j++)
        L.data[j-1]=L.data[j];
    L.length--;
    return true;
}
```

### 顺序查找

```c++
int LocateElem(SqList L, int e){
		int i;
    for(i=0; i<L.length; i++)
       if(L.data[i] == e)
          return i+1;
    return 0;
}
```

## 线性表的链式表示
### 单链表结点类型的定义
```c++
  typedef struct LNode{
    int data;
    struct LNode *next;   
  }LNode, *LinkList;
```

### 头插法建立单链表
```c++
  LinkList List_HeadInsert(LinkList &L){
    LNode *s; 
    int x;
    L=(LinkList)malloc(siezof(LNode));
    L->next=NULL;
    scanf("%d", &x);
    while (x!=9999) // 输入9999表示结束
    {
      s=(LNode*)malloc(sizeof(LNode));
      s->data=x;
      s->next=L->next;
      L->next=s;
      scanf("%d", &x);
    }
    return L;
  }
```

### 尾插法建立单链表
```c++
  LinkList List_TailInsert(LinkList & L) {
      int x;
      L = (LinkList)malloc(siezof(LNode));
      LNode* s, *r=L;   // r为表尾指针
      scanf("%d", &x);
      while (x != 9999)  // 输入9999表示结束
      {
          s = (LNode*)malloc(sizeof(LNode));
          s->data = x;
          r->next=s;
          r=s;  // r指向新的表尾结点
          scanf("%d", &x);
      }
      r->next=NULL;  // 尾结点指针置空
      return L;
  }
```

### 按顺序查找结点值
```c++
  Lnode *GetElem(LinkList L, int i){
    int j=1;
    LNode *p=L->next;
    if (i==0)
      return L;
    if(i<1)
      return NULL;
    while (p&&j<i)
    {
      p=p->next;
      j++;
    }
    return p;
  }
```

### 按值查找表结点
```c++
  LNode* LocateElem(LinkList L, int e) {
      LNode* p = L->next;
      while (p != NULL && p->data!=e) {
          p = p->next;
      }
      return p;
  }
```

### 插入结点操作

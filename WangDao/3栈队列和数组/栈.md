  }

  bool Pop(SqStack &S, int x) {
      if (S.top == 1)
          return false;
      x = S.data[S.top--];
      return true;
  }

  bool GetTop(SqStack &S, int x) {
      if (S.top == -1)
          return false;
      x = S.data[S.top];
      return true;
  }

### 定长顺序存储表示
```c++
typedef struct {
    char ch[100];
    int length;
} SString;
```

### 堆分配存储表示
```c++
typedef struct {
    char *ch; // 按串长分配存储区，ch指向串的基地址
    int length;
} HString;
```

### 模式匹配
```c++
// 朴素模式匹配
    int Index(SString S, SString T){
      int i = 1, j= 1;
      while (i<=S.length && j <=T.length){
        if (S.ch[i] == T.ch[j])
          ++i; ++j;
        else
          i=i-j+2; j=1;
      }
      if (j>T.length)
        return i-T.length;
      else
        return 0;
    }

    void get_next(String T, int next[]){
      int i = 1; j = 0; 
      next[1]=0;
      while (i<T.length) {
        if (j==0 || T.ch[i]==T.ch[j]) {
          ++i;
          ++j;
          next[i]=j;
        }
        else 
          j=next[j];
      }
    }
```
